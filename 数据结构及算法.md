#### 两数之和

```go
func TwoSum(nums []int, target int) []int {
  	// 引入一个hashmap,用于保存已遍历数据，v=>k
  	tmp := map[int]int{}
    for i := 0; i < len(nums); i++ {
        t := target - nums[i];
        if _, ok := tmp[t]; ok {
          return []int{tmp[t], i}
        } else {
          tmp[nums[i]] = i
        }
      }
    return nil
}
```

#### 合并两个有序数组

```go
// 给你两个按非递减顺序排列的整数数组，nums1和nums2
func mergeArr(nums1 []int, nums2 []int) []int {
  i, j := 0,0 // 双指针，分别指向num1和num2的头部
  tmp := []int{} // 结果数组
  for i < len(nums1) && j < len(nums2) {
    if nums1[i] > nums2[j] {
      tmp = append(tmp, nums2[j])
      j++
    } else {
      tmp = append(tmp, nums2[i])
      i++
    }
  }
  if i < len(nums1) {
    tmp = append(tmp, nums1...)
  }
  
  if j < len(nums2) {
    tmp = append(tmp, nums2...)
  }
  
  return tmp
}
```

#### 斐波那契数列

写一个函数，输入n,求斐波那契数列的第n项。斐波那契数列的定义如下

```go
f(n) : 0 => n = 0
			 1 => n = 1
			 f(n-2) + f(n-1) n > 1

// 利用递归求解
var map[int]int // 用于记录计算的值
func fb2(n int) int {
  if n <= 1 {
    return n
  }
  
  if v, ok := map[n]; ok {
    return v
  }
  
  r := f(n-2) + f(n-1)
  map[n] = r
  
  return r
}


// 循环方式求解
func fb(n int) int {
  if n <= 1 {
    return n
  }
  
  result := 0;
  prePre := 0;
  pre := 1;
  for i := 2; i < n; i++ {
    result = prePre + pre
    prePre = pre
    pre = result
  }
  
  return result
}
```
